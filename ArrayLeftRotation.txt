#include <math.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <assert.h>
#include <limits.h>
#include <stdbool.h>

void leftRotate(int arr[], int d, int n)
{
  int i;
  for (i = 0; i < d; i++)
    leftRotatebyOne(arr, n);
}
 
void leftRotatebyOne(int arr[], int n)
{
  int i, temp;
  temp = arr[0];
  for (i = 0; i < n-1; i++)
  arr[i] = arr[i+1];
  arr[i] = temp;
}
 
void printArray(int arr[], int size)
{
  int i;
  for(i = 0; i < size; i++)
  printf("%d ", arr[i]);
  printf("\n");
}
 
int main()
{
    int n,d;
   scanf("%d %d",&n,&d);
   int arr[n];
   for(int i = 0; i < n; i++)
   scanf("%d ", &arr[i]);
   leftRotate(arr, d, n);
   printArray(arr, n);
   getchar();
   return 0;
}
